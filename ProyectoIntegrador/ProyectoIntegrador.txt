CREATE TABLE Proveedor (CIF varchar2(9) CONSTRAINT CIF_pk PRIMARY KEY,
                        Nombre varchar2(25),
                        Correo varchar2(50),
                        TFNO number(9),
                        Direc varchar2(35),
                        Cod_po number(10),
                        Des varchar2(50));
CREATE TABLE Pedidos (Cod_pd number(5) CONSTRAINT Cod_pd_pk PRIMARY KEY,
                      Fecha DATE,
                      Tot_importe number(6,2),
                      CIF varchar2(9),
                      CONSTRAINT CIF_fk FOREIGN KEY(CIF) REFERENCES Proveedor(CIF));
CREATE TABLE Material (Cod_mt number(5) CONSTRAINT Cod_mt_pk PRIMARY KEY,
                       Nombre varchar2(25),
                       Des varchar2(50),
                       Tipo varchar2(20),
                       Stock number(5));
CREATE TABLE Proyectos (Cod_pr number(5) CONSTRAINT Cod_pr_pk PRIMARY KEY,
                        Nombre varchar2(50),
                        Des varchar2(50));
CREATE TABLE Maquina (Cod_ma number(5) CONSTRAINT Cod_ma_pk PRIMARY KEY,
                      Nombre varchar2(25),
                      Des varchar2(50),
                      Tipo varchar2(20));
CREATE TABLE Usuarios (ID_Soci number(5) CONSTRAINT ID_Soci_pk PRIMARY KEY,
                       Nombre varchar2(25),
                       Apell varchar2(25),
                       Correo varchar2(30),
                       Tfno number(9),
                       Tipo_us varchar2(25));
CREATE TABLE Eventos (Cod_ev number(5) CONSTRAINT Cod_ev_pk PRIMARY KEY,
                      Fecha DATE,
                      Mentor varchar2(30),
                      Categoria varchar2(30),
                      Duracion varchar2(30),
                      Lugar varchar2(30));
CREATE TABLE Linea_Pedido (Cod_pd number(5),
                           Cod_mt number(5),
                           Cantidad number(5),
                           Precio_unidad number(6,2),
                           CONSTRAINT Linea_pedido_pk PRIMARY KEY (Cod_pd, Cod_mt),
                           CONSTRAINT Cod_pd_Linea_pedido_fk FOREIGN KEY(Cod_pd) REFERENCES Pedidos(Cod_pd),
                           CONSTRAINT Cod_mt_Linea_pedido_fk FOREIGN KEY(Cod_mt) REFERENCES Material(Cod_mt));
CREATE TABLE MaterialProyectos (Cod_mt number(5),
                                Cod_pr number(5),
                                Cantidad number(5),
                                CONSTRAINT MaterialProyectos_pk PRIMARY KEY (Cod_mt, Cod_pr),
                                CONSTRAINT Cod_mt_MaterialProyectos_fk FOREIGN KEY(Cod_mt) REFERENCES Material(Cod_mt),
                                CONSTRAINT Cod_pr_MaterialProyectos_fk FOREIGN KEY(Cod_pr) REFERENCES Proyectos(Cod_pr));
CREATE TABLE ProyectosMaquina (Cod_pr number(5),
                               Cod_ma number(5),
                               Fecha_inicio DATE,
                               Fecha_fin DATE,
                               CONSTRAINT ProyectosMaquina_pk PRIMARY KEY (Cod_pr, Cod_ma),
                               CONSTRAINT Cod_pr_ProyectosMaquina_fk FOREIGN KEY(Cod_pr) REFERENCES Proyectos(Cod_pr),
                               CONSTRAINT Cod_ma_ProyectosMaquina_fk FOREIGN KEY(Cod_ma) REFERENCES Maquina(Cod_ma));
CREATE TABLE ProyectosUsuarios (Cod_pr number(5),
                                ID_Soci number(5),
                                CONSTRAINT ProyectosUsuarios_pk PRIMARY KEY (Cod_pr, ID_Soci),
                                CONSTRAINT Cod_pr_ProyectosUsuarios_fk FOREIGN KEY(Cod_pr) REFERENCES Proyectos(Cod_pr),
                                CONSTRAINT ID_Soci_ProyectosUsuarios_fk FOREIGN KEY(ID_Soci) REFERENCES Usuarios(ID_Soci));
CREATE TABLE UsuariosEventos (ID_Soci number(5),
                              Cod_ev number(5),
                              CONSTRAINT UsuariosEventos_pk PRIMARY KEY (ID_Soci, Cod_ev),
                              CONSTRAINT ID_Soci_UsuariosEventos_fk FOREIGN KEY(ID_Soci) REFERENCES Usuarios(ID_Soci),
                              CONSTRAINT Cod_ev_UsuariosEventos_fk FOREIGN KEY(Cod_ev) REFERENCES Eventos(Cod_ev));
